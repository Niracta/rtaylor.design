## Lists

### Unordered Lists

Groups a collection of items that do not have a numerical ordering, and their order in the list is meaningless.

- Donec non tortor in arcu mollis feugiat
- Lorem ipsum dolor sit amet, consectetuer adipiscing elit
- Donec id eros eget quam aliquam gravida
- Vivamus convallis urna id felis
- Nulla porta tempus sapien

### Ordered Lists

  ### Text Inputs

  <Form
    baseId="text-inputs"
    autoComplete="new-password"
    values={{
      passive: ``,
      active: `This is a Valid Value`,
      error: `This is an Invalid Value`,
      disabled: ``
    }}
    validateOnBlur
    validateOnChange
    onValidate={values =>
      object({
        passive: string(),
        active: string(),
        error: number(),
        disabled: string(),
      }).validate(values, { abortEarly: false }).then(
        () => {},
        error => {
          if (error.inner.length) {
            throw error.inner.reduce(
              (errors, curr) =>
                set(errors, curr.path, curr.message),
              {}
            );
          }
        }
      )
    }
  >
    <Input type="text" data-lpignore="true" name="passive" placeholder="Passive Form Field - Montserrat Italic - 28px" />
    <Input type="text" data-lpignore="true" name="active" placeholder="Active form field • Montserrat Italic • 28px" />
    <Input type="url" required data-lpignore="true" name="error" placeholder="Error Form Field • Montserrat Italic • 28px" />
    <Input type="text" disabled data-lpignore="true" name="disabled" placeholder="Disabled Form field • Montserrat Italic • 28px" />
  </Form>

  ### Checkboxes

  <Form
    baseId="checkboxes"
    values={{
      passive: false,
      active: true,
      error: undefined,
      disabled: false,
      checkboxGroup: [`active`]
    }}
    validateOnBlur
    validateOnChange
    onValidate={values => {
      const errors = {};
      if (!values.error) {
        errors.error = "You must accept our not-so-evil conditions!";
      }
      if (Object.keys(errors).length) {
        throw errors;
      }
    }}
  >
    <Group>
      <Checkbox name="passive">Passive</Checkbox>
      <Checkbox name="active">Active</Checkbox>
      <Checkbox name="error">Error</Checkbox>
      <Checkbox disabled name="disabled">Disabled</Checkbox>
    </Group>
    <Group name="checkboxGroup">
      <Checkbox value="passive">Passive</Checkbox>
      <Checkbox value="active">Active</Checkbox>
      <Checkbox value="error">Error</Checkbox>
      <Checkbox disabled value="disabled">Disabled</Checkbox>
    </Group>
  </Form>

  ### Radio Buttons

  <Form
    baseId="radios"
    values={{
      radioGroup: `active`
    }}
    validateOnBlur
    validateOnChange
    onValidate={values => {
      if (values.radioGroup === "error") {
        const errors = { radioGroup: "Invalid choice!" };
        throw errors;
      }
    }}
  >
    <RadioGroup name="radioGroup">
      <Radio value="passive">Passive</Radio>
      <Radio value="active">Active</Radio>
      <Radio value="error">Error</Radio>
      <Radio disabled value="disabled">Disabled</Radio>
    </RadioGroup>
  </Form>
